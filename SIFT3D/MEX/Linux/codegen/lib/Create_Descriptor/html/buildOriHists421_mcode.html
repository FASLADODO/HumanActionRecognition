1function myhist = buildOriHists(x,y,z,radius,pix,fv)
 2% LoadParams;
 3% lData = load('LoadParams.mat');
 4% 
 5% Display_flag = lData.Display_flag;
 6% IgnoreGradSign = lData.IgnoreGradSign;
 7% IndexSigma = lData.IndexSigma;
 8% IndexSize = lData.IndexSize;
 9% MagFactor = lData.MagFactor;
10% nFaces = lData.nFaces;
11% OriHistThresh = lData.OriHistThresh;
12% OriSigma = lData.OriSigma;
13% Smooth_Flag = lData.Smooth_Flag;
14% Smooth_Var = lData.Smooth_Var;
15% Tessellation_flag = lData.Tessellation_flag;
16% Tessellation_levels = lData.Tessellation_levels;
17% TwoPeak_Flags = lData.TwoPeak_Flag;
18% UseHistogramOri = lData.UseHistogramOri;
19
20
21TwoPeak_Flag = 1;  % Allow 3DSIFT to throw out points, Default: 1
22IndexSize = 2;  % Min: 1  Default: 2 
23
24Display_flag = 0;  % Display a sphere which can be rotated (Rotate 3D button) to view gradient directions
25Tessellation_flag = 1;  % Keep this as 1
26Tessellation_levels = 1;  % Min: zero  Default: 1
27nFaces = 20 * ( 4 ^ Tessellation_levels );  % Number of faces in the tessellation, not a parameter
28
29Smooth_Flag = 1;  % Adds gradient data to surrounding bins in final histogram
30Smooth_Var = 20;  % Determines amount of smoothing, Default: 20
31
32% The rest of the variables are not modified often, but are included here
33% for completeness sake
34IgnoreGradSign = 0;
35IndexSigma = 5.0;
36MagFactor = 3;
37
38UseHistogramOri = 1;
39OriHistThresh = 0.8;
40if (UseHistogramOri)
41    OriSigma = 1.5;
42else
43    OriSigma = 1.0;
44end
45
46
47xi = int16(x);
48yi = int16(y);
49zi = int16(z);
50[rows, cols, slices] = size(pix);
51myhist = zeros(1,nFaces);
52
53
54for r = xi - radius:xi + radius
55    for c = yi - radius:yi + radius
56        for s = zi - radius:zi + radius
57            %          /* Do not use last row or column, which are not valid. */
58            if (r >= 1 && c >= 1 && r < rows - 1 && c < cols - 1 && s >= 1 && s < slices - 1)
59                [mag, vect] = GetGradOri_vector(pix, r, c, s);
60                
61                corr_array = fv.centers * vect';
62
63                % [yy, ix] = sort(corr_array,'descend');
64                [~, ix] = sort(corr_array, 'descend');
65                
66                if (Display_flag ==1)
67                    plot3(vect(1), vect(2), vect(3), 'k.')
68                    plot3(fv.centers(ix(1), 1), fv.centers(ix(1), 2), fv.centers(ix(1), 3), 'b*')
69                end
70                
71                myhist(ix(1)) = myhist(ix(1)) + mag;
72            end
73        end
74    end
75end
