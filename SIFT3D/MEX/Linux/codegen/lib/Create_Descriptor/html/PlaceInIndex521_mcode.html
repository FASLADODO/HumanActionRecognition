1function index = PlaceInIndex(index, mag, vect, i, j, s, fv)
 2
 3% LoadParams;
 4% lData = load('LoadParams');
 5% 
 6% Display_flag = lData.Display_flag;
 7% IgnoreGradSign = lData.IgnoreGradSign;
 8% IndexSigma = lData.IndexSigma;
 9% IndexSize = lData.IndexSize;
10% MagFactor = lData.MagFactor;
11% nFaces = lData.nFaces;
12% OriHistThresh = lData.OriHistThresh;
13% OriSigma = lData.OriSigma;
14% Smooth_Flag = lData.Smooth_Flag;
15% Smooth_Var = lData.Smooth_Var;
16% Tessellation_flag = lData.Tessellation_flag;
17% Tessellation_levels = lData.Tessellation_levels;
18% TwoPeak_Flags = lData.TwoPeak_Flag;
19% UseHistogramOri = lData.UseHistogramOri;
20TwoPeak_Flag = 1;  % Allow 3DSIFT to throw out points, Default: 1
21IndexSize = 2;  % Min: 1  Default: 2 
22
23Display_flag = 0;  % Display a sphere which can be rotated (Rotate 3D button) to view gradient directions
24Tessellation_flag = 1;  % Keep this as 1
25Tessellation_levels = 1;  % Min: zero  Default: 1
26nFaces = 20 * ( 4 ^ Tessellation_levels );  % Number of faces in the tessellation, not a parameter
27
28Smooth_Flag = 1;  % Adds gradient data to surrounding bins in final histogram
29Smooth_Var = 20;  % Determines amount of smoothing, Default: 20
30
31% The rest of the variables are not modified often, but are included here
32% for completeness sake
33IgnoreGradSign = 0;
34IndexSigma = 5.0;
35MagFactor = 3;
36
37UseHistogramOri = 1;
38OriHistThresh = 0.8;
39if (UseHistogramOri)
40    OriSigma = 1.5;
41else
42    OriSigma = 1.0;
43end
44corr_array = fv.centers * vect';
45
46[yy, ix] = sort(corr_array,'descend');
47
48if (Smooth_Flag == 0)
49    index(i,j,s,ix(1)) = index(i,j,s,ix(1)) + mag;
50elseif (Smooth_Flag == 1)
51    tmpsum = sum(yy(1:3).^Smooth_Var);
52    for ii=1:3
53        index(i,j,s,ix(ii)) = index(i,j,s,ix(ii)) + ( mag * ( yy(ii) .^ Smooth_Var ) / tmpsum );
54    end
55end
