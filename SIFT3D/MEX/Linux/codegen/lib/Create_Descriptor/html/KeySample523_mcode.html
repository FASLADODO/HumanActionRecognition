1function index = KeySample(key, pix)
  2
  3% LoadParams;
  4% lData = load('LoadParams.mat');
  5% 
  6% Display_flag = lData.Display_flag;
  7% IgnoreGradSign = lData.IgnoreGradSign;
  8% IndexSigma = lData.IndexSigma;
  9% IndexSize = lData.IndexSize;
 10% MagFactor = lData.MagFactor;
 11% nFaces = lData.nFaces;
 12% OriHistThresh = lData.OriHistThresh;
 13% OriSigma = lData.OriSigma;
 14% Smooth_Flag = lData.Smooth_Flag;
 15% Smooth_Var = lData.Smooth_Var;
 16% Tessellation_flag = lData.Tessellation_flag;
 17% Tessellation_levels = lData.Tessellation_levels;
 18% TwoPeak_Flags = lData.TwoPeak_Flag;
 19% UseHistogramOri = lData.UseHistogramOri;
 20TwoPeak_Flag = 1;  % Allow 3DSIFT to throw out points, Default: 1
 21IndexSize = 2;  % Min: 1  Default: 2 
 22
 23Display_flag = 0;  % Display a sphere which can be rotated (Rotate 3D button) to view gradient directions
 24Tessellation_flag = 1;  % Keep this as 1
 25Tessellation_levels = 1;  % Min: zero  Default: 1
 26nFaces = 20 * ( 4 ^ Tessellation_levels );  % Number of faces in the tessellation, not a parameter
 27
 28Smooth_Flag = 1;  % Adds gradient data to surrounding bins in final histogram
 29Smooth_Var = 20;  % Determines amount of smoothing, Default: 20
 30
 31% The rest of the variables are not modified often, but are included here
 32% for completeness sake
 33IgnoreGradSign = 0;
 34IndexSigma = 5.0;
 35MagFactor = 3;
 36
 37UseHistogramOri = 1;
 38OriHistThresh = 0.8;
 39if (UseHistogramOri)
 40    OriSigma = 1.5;
 41else
 42    OriSigma = 1.0;
 43end
 44
 45fv = sphere_tri('ico', Tessellation_levels, 1, 0);
 46
 47
 48irow = int16(key.x); % x
 49icol = int16(key.y); % y
 50islice = int16(key.z); % t
 51
 52xySpacing = key.xyScale * MagFactor;
 53tSpacing = key.tScale * MagFactor;
 54
 55xyRadius = 1.414 * xySpacing * (IndexSize + 1) / 2.0;
 56tRadius = 1.414 * tSpacing * (IndexSize + 1) / 2.0;
 57xyiradius = int16(xyRadius);
 58tiradius = int16(tRadius);
 59
 60index = zeros(IndexSize,IndexSize,IndexSize,nFaces);
 61
 62for i = -xyiradius : xyiradius
 63    for j = -xyiradius : xyiradius
 64        for s = -tiradius : tiradius
 65
 66            % This is redundant and probably slows down the code, but at
 67            % some point this solved a major overflow headache, so leaving
 68            % as-is for now
 69            i2 = double(i);
 70            j2 = double(j);
 71            s2 = double(s);
 72            distsq = double(i2^2 + j2^2 + s2^2);
 73
 74            v0 = [i2; j2; s2];
 75
 76            i_indx = int16(floor(double((i + xyiradius)) / double((2*xyiradius/IndexSize)))) + 1;
 77            j_indx = int16(floor(double((j + xyiradius)) / double((2*xyiradius/IndexSize)))) + 1;
 78            s_indx = int16(floor(double((s + tiradius )) / double((2*tiradius/IndexSize )))) + 1;
 79            
 80            if i_indx > IndexSize
 81                % i_indx = IndexSize;
 82                i_indx = cast(IndexSize,'int16');
 83            end
 84            if j_indx > IndexSize
 85                % j_indx = IndexSize;
 86                j_indx = cast(IndexSize,'int16');
 87            end
 88            if s_indx > IndexSize
 89                %s_indx = IndexSize;
 90                s_indx = cast(IndexSize,'int16');
 91            end
 92
 93            if (i_indx < 1 || j_indx < 1 || s_indx < 1)
 94                disp('Something wrong with the sub-histogram index');
 95            end
 96            
 97           
 98            r = irow + v0(1);
 99            c = icol + v0(2);
100            t = islice + v0(3);
101            
102            index = AddSample(index, key, pix, distsq, r, c, t, i_indx, j_indx, s_indx, fv);
103            
104        end
105    end
106end
