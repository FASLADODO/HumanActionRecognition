1function [keyPoint, reRun] = Create_Descriptor(pix, xyScale, tScale, x, y, z)
 2
 3% Main function of 3DSIFT Program from http://www.cs.ucf.edu/~pscovann/
 4%
 5% Inputs:
 6% pix - a 3 dimensional matrix of uint8
 7% xyScale and tScale - affects both the scale and the resolution, these are
 8% usually set to 1 and scaling is done before calling this function
 9% x, y, and z - the location of the center of the keypoint where a descriptor is requested
10%
11% Outputs:
12% keypoint - the descriptor, varies in size depending on values in LoadParams.m
13% reRun - a flag (0 or 1) which is set if the data at (x,y,z) is not
14% descriptive enough for a good keypoint
15%
16% Example:
17% See Demo.m
18
19% LoadParams is a .mat files. To modify the parameters, open
20% LoadParamsBkp.m and change parameters.
21% lData = load('LoadParams.mat');
22% 
23% Display_flag = lData.Display_flag;
24% IgnoreGradSign = lData.IgnoreGradSign;
25% IndexSigma = lData.IndexSigma;
26% IndexSize = lData.IndexSize;
27% MagFactor = lData.MagFactor;
28% nFaces = lData.nFaces;
29% OriHistThresh = lData.OriHistThresh;
30% OriSigma = lData.OriSigma;
31% Smooth_Flag = lData.Smooth_Flag;
32% Smooth_Var = lData.Smooth_Var;
33% Tessellation_flag = lData.Tessellation_flag;
34% Tessellation_levels = lData.Tessellation_levels;
35% TwoPeak_Flag = lData.TwoPeak_Flag;
36% UseHistogramOri = lData.UseHistogramOri;

38TwoPeak_Flag = 1;  % Allow 3DSIFT to throw out points, Default: 1
39IndexSize = 2;  % Min: 1  Default: 2 
40
41Display_flag = 0;  % Display a sphere which can be rotated (Rotate 3D button) to view gradient directions
42Tessellation_flag = 1;  % Keep this as 1
43Tessellation_levels = 1;  % Min: zero  Default: 1
44nFaces = 20 * ( 4 ^ Tessellation_levels );  % Number of faces in the tessellation, not a parameter
45
46Smooth_Flag = 1;  % Adds gradient data to surrounding bins in final histogram
47Smooth_Var = 20;  % Determines amount of smoothing, Default: 20

49% The rest of the variables are not modified often, but are included here
50% for completeness sake
51IgnoreGradSign = 0;
52IndexSigma = 5.0;
53MagFactor = 3;
54
55UseHistogramOri = 1;
56OriHistThresh = 0.8;
57if (UseHistogramOri)
58    OriSigma = 1.5;
59else
60    OriSigma = 1.0;
61end

63
64
65ivec = zeros(1, (IndexSize*IndexSize*IndexSize*nFaces));
66
67keyPoint = struct('x',x,'y',y, 'z', z,'xyScale', xyScale, 'tScale', tScale,'ivec', ivec);
68
69
70reRun = 0;
71
72radius = int16(xyScale * 3.0);
73
74fv = sphere_tri('ico',Tessellation_levels,1,0);
75myhist = buildOriHists(x,y,z,radius,pix,fv);
76
77% [yy, ix] = sort(myhist,'descend');

79% Dom_Peak = ix(1);
80% Sec_Peak = ix(2);
81
82if (TwoPeak_Flag == 1 && dot(fv.centers(ix(1),:),fv.centers(ix(2),:)) > .9 && dot(fv.centers(ix(1),:),fv.centers(ix(3),:)) > .9)
83    disp('MISS : Top 3 orientations within ~25 degree range : Returning with reRun flag set.');
84    reRun = 1;
85    return;
86end
