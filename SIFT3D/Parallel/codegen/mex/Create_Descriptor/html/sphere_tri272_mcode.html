1function [FV] = sphere_tri(shape, maxlevel, r, winding)
  2coder.varsize('vertices');
  3coder.varsize('faces');
  4
  5% sphere_tri - generate a triangle mesh approximating a sphere
  6%
  7% Usage: FV = sphere_tri(shape,Nrecurse,r,winding)
  8%
  9%   shape is a string, either of the following:
 10%   'ico'   starts with icosahedron (most even, default)
 11%   'oct'   starts with octahedron
 12%   'tetra' starts with tetrahedron (least even)
 13%
 14%   Nrecurse is int >= 0, setting the recursions (default 0)
 15%
 16%   r is the radius of the sphere (default 1)
 17%
 18%   winding is 0 for clockwise, 1 for counterclockwise (default 0).  The
 19%   matlab patch command gives outward surface normals for clockwise
 20%   order of vertices in the faces (viewed from outside the surface).
 21%
 22%   FV has fields FV.vertices and FV.faces.  The vertices
 23%   are listed in clockwise order in FV.faces, as viewed
 24%   from the outside in a RHS coordinate system.
 25%
 26% The function uses recursive subdivision.  The first
 27% approximation is an platonic solid, either an  icosahedron,
 28% octahedron or a tetrahedron.  Each level of refinement
 29% subdivides each triangle face by a factor of 4 (see also
 30% mesh_refine).  At each refinement, the vertices are
 31% projected to the sphere surface (see sphere_project).
 32%
 33% A recursion level of 3 or 4 is a good sphere surface, if
 34% gouraud shading is used for rendering.
 35%
 36% The returned struct can be used in the patch command, eg:
 37%
 38% % create and plot, vertices: [2562x3] and faces: [5120x3]
 39% FV = sphere_tri('ico',4,1);
 40% lighting phong; shading interp; figure;
 41% patch('vertices',FV.vertices,'faces',FV.faces,...
 42%       'facecolor',[1 0 0],'edgecolor',[.2 .2 .6]);
 43% axis off; camlight infinite; camproj('perspective');
 44%
 45% See also: mesh_refine, sphere_project
 46%
 47
 48
 49
 50% $Revision: 1.2 $ $Date: 2005/07/20 23:07:03 $
 51
 52% Licence:  GNU GPL, no implied or express warranties
 53% Jon Leech (leech @ cs.unc.edu) 3/24/89
 54% icosahedral code added by Jim Buddenhagen (jb1556@daditz.sbc.com) 5/93
 55% 06/2002, adapted from c to matlab by Darren.Weber_at_radiology.ucsf.edu
 56% 05/2004, reorder of the faces for the 'ico' surface so they are indeed
 57% clockwise!  Now the surface normals are directed outward.  Also reset the
 58% default recursions to zero, so we can get out just the platonic solids.
 59%
 60%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
 61
 62%eegversion = '$Revision: 1.2 $';
 63% fprintf('SPHERE_TRI [v %s]\n',eegversion(11:15)); tic
 64%{
 65if size(varargin,2) < 1
 66    shape = 'ico';
 67    maxlevel = 0;
 68    r = 1;
 69    winding = 0;
 70elseif size(varargin,2) < 2
 71    maxlevel = 0;
 72    r = 1;
 73    winding = 0;
 74elseif size(varargin,2) < 3
 75    r = 1;
 76    winding = 0;
 77else
 78    winding = 0;
 79end
 80%}
 81%{
 82if ~exist('shape','var') || isempty(shape),
 83    shape = 'ico';
 84end
 85% fprintf('...creating sphere tesselation based on %s\n',shape);
 86
 87% default maximum subdivision level
 88if ~exist('maxlevel','var') || isempty(maxlevel) || maxlevel < 0,
 89    maxlevel = 0;
 90end
 91
 92% default radius
 93if ~exist('r','var') || isempty(r),
 94    r = 1;
 95end
 96
 97if ~exist('winding','var') || isempty(winding),
 98    winding = 0;
 99end
100%}
101
102
103% -----------------
104% define the starting shapes
105
106shape = lower(shape);
107
108switch shape,
109case 'tetra',
110    
111    % Vertices of a tetrahedron
112    sqrt_3 = 0.5773502692;
113    
114    tetra.v = [  sqrt_3,  sqrt_3,  sqrt_3 ;   % +X, +Y, +Z  - PPP
115                -sqrt_3, -sqrt_3,  sqrt_3 ;   % -X, -Y, +Z  - MMP
116                -sqrt_3,  sqrt_3, -sqrt_3 ;   % -X, +Y, -Z  - MPM
117                 sqrt_3, -sqrt_3, -sqrt_3 ];  % +X, -Y, -Z  - PMM
118    
119    % Structure describing a tetrahedron
120    tetra.f = [ 1, 2, 3;
121                1, 4, 2;
122                3, 2, 4;
123                4, 1, 3 ];
124    
125    vertices = tetra.v;
126    faces    = tetra.f;
127    
128case 'oct',
129    oct = struct('v',zeros(6,3),'f',zeros(8,3));
130    % Six equidistant points lying on the unit sphere
131    oct.v = [  1,  0,  0 ;  %  X
132              -1,  0,  0 ;     % -X
133               0,  1,  0 ;  %  Y
134               0, -1,  0 ;     % -Y
135               0,  0,  1 ;     %  Z
136               0,  0, -1 ];    % -Z
137    
138    % Join vertices to create a unit octahedron
139    oct.f = [ 1 5 3 ;    %  X  Z  Y  -  First the top half
140              3 5 2 ;    %  Y  Z -X
141              2 5 4 ;    % -X  Z -Y
142              4 5 1 ;    % -Y  Z  X
143              1 3 6 ;    %  X  Y -Z  -  Now the bottom half
144              3 2 6 ;    %  Y  Z -Z
145              2 4 6 ;    % -X  Z -Z
146              4 1 6 ];   % -Y  Z -Z
147    
148    vertices = oct.v;
149    faces    = oct.f;
150    
151case 'ico',
152    
153    % Twelve vertices of icosahedron on unit sphere
154    tau = 0.8506508084; % t=(1+sqrt(5))/2, tau=t/sqrt(1+t^2)
155    one = 0.5257311121; % one=1/sqrt(1+t^2) , unit sphere
156    ico = struct('v',zeros(12,3),'f',zeros(20,3));
157    ico.v( 1,:) = [  tau,  one,    0 ]; % ZA
158    ico.v( 2,:) = [ -tau,  one,    0 ]; % ZB
159    ico.v( 3,:) = [ -tau, -one,    0 ]; % ZC
160    ico.v( 4,:) = [  tau, -one,    0 ]; % ZD
161    ico.v( 5,:) = [  one,   0 ,  tau ]; % YA
162    ico.v( 6,:) = [  one,   0 , -tau ]; % YB
163    ico.v( 7,:) = [ -one,   0 , -tau ]; % YC
164    ico.v( 8,:) = [ -one,   0 ,  tau ]; % YD
165    ico.v( 9,:) = [   0 ,  tau,  one ]; % XA
166    ico.v(10,:) = [   0 , -tau,  one ]; % XB
167    ico.v(11,:) = [   0 , -tau, -one ]; % XC
168    ico.v(12,:) = [   0 ,  tau, -one ]; % XD
169    
170    % Structure for unit icosahedron
171    ico.f = [  5,  8,  9 ;
172               5, 10,  8 ;
173               6, 12,  7 ;
174               6,  7, 11 ;
175               1,  4,  5 ;
176               1,  6,  4 ;
177               3,  2,  8 ;
178               3,  7,  2 ;
179               9, 12,  1 ;
180               9,  2, 12 ;
181              10,  4, 11 ;
182              10, 11,  3 ;
183               9,  1,  5 ;
184              12,  6,  1 ;
185               5,  4, 10 ;
186               6, 11,  4 ;
187               8,  2,  9 ;
188               7, 12,  2 ;
189               8, 10,  3 ;
190               7,  3, 11 ];
191    
192    vertices = ico.v;
193    faces    = ico.f;
194end
195
196
197% -----------------
198% refine the starting shapes with subdivisions
199if maxlevel,
200    
201    % Subdivide each starting triangle (maxlevel) times
202    for level = 1:maxlevel,
203        
204        % Subdivide each triangle and normalize the new points thus
205        % generated to lie on the surface of a sphere radius r.
206        [vertices, faces] = mesh_refine_tri4(vertices, faces);
207        % FV.vertices = sphere_project(FV.vertices,r);
208        vertices = sphere_project(vertices, r, [0,0,0]);
209        
210        % An alternative might be to define a min distance
211        % between vertices and recurse or use fminsearch
212        
213    end
214end
215
216centers = zeros(size(faces));
217
218for i=1:length(faces)
219    centers(i,:) = mean(vertices(faces(i,:),:));
220    % Unit Normalization
221    centers(i,:) = centers(i,:) ./ sqrt(dot(centers(i,:),centers(i,:)));
222end
223
224if winding
225%     fprintf('...returning counterclockwise vertex order (viewed from outside)\n');
226    faces = faces(:,[1 3 2]);
227else
228%     fprintf('...returning clockwise vertex order (viewed from outside)\n');
229end
230
231% t=toc; fprintf('...done (%6.2f sec)\n\n',t);
232FV = struct('vertices',vertices, 'faces',faces,'centers',centers);
233return
